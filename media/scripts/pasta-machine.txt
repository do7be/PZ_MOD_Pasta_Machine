module PastaMachine
{
    imports
    {
        Base
    }

	item PastaMachine
	{
		Weight	=	1,
		Type	=	Normal,
		DisplayName	=	Pasta Machine,
		Icon	=	Bowl,
	}

    recipe Make Pasta
    {
       keep PastaMachine.PastaMachine,
       Water=2,
       Flour=2,
       Salt;1,

       Result:PastaMachine.RawPasta,
       Time:50.0,
       Category:Cooking,
       OnGiveXP:Recipe.OnGiveXP.Cooking10,
    }

    item RawPasta
    {
        Weight	=	2,
        Type	=	Food,
        DisplayName	=	Raw Pasta,
        Icon	=	SpagettiRaw,
        HungerChange    =   -60,
        UnhappyChange   =   40,
        ThirstChange    =   60,
        Carbohydrates = 656,
        Proteins = 112,
        Lipids = 16,
        Calories = 3360,
        WorldStaticModel = Pasta,
        ReplaceOnCooked     =   Base.Pasta,
        IsCookable	=	    true,
		DangerousUncooked	=	TRUE,
        MinutesToCook   =   20,
        DaysFresh 		= 	4,
        DaysTotallyRotten 	= 	7,
        BadInMicrowave = true,
        EvolvedRecipe = PastaPot:10,
    }

    recipe Place Raw Pasta in Cooking Pot
	{
	   PastaMachine.RawPasta;10,
	   destroy WaterPot,

	   Result:WaterPotPasta,
	   Time:50.0,
	   OnCreate:Recipe.OnCreate.AddBaseIngredientToCookingVessel,
	   Category:Cooking,
	   OnGiveXP:Recipe.OnGiveXP.None,
	}
}
