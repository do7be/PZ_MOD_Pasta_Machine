module PastaMachine
{
    imports
    {
        Base
    }

	item PastaMachine
	{
		Weight	=	1,
		Type	=	Normal,
		DisplayName	=	Pasta Machine,
		Icon	=	PastaMachine,
	}

    recipe Make Fresh Pasta
    {
       keep PastaMachine.PastaMachine,
       Water=2,
       Flour=2,
       Salt;1,

       Result:PastaMachine.FreshPasta,
       Time:50.0,
       Category:Cooking,
       OnGiveXP:Recipe.OnGiveXP.Cooking10,
    }

    item FreshPasta
    {
        Weight	=	2,
        Type	=	Food,
        DisplayName	=	Fresh Pasta,
        Icon	=	SpagettiRaw,
        HungerChange    =   -60,
        UnhappyChange   =   40,
        ThirstChange    =   60,
        Carbohydrates = 656,
        Proteins = 112,
        Lipids = 16,
        Calories = 3360,
        WorldStaticModel = Pasta,
        /* TODO: 生パスタを消費した状態で加熱すると量が満タンのパスタになるがどうしようもなさそう */
        ReplaceOnCooked     =   Base.Pasta,
        IsCookable	=	    true,
		DangerousUncooked	=	TRUE,
        MinutesToCook   =   20,
        DaysFresh 		= 	4,
        DaysTotallyRotten 	= 	7,
        BadInMicrowave = true,
        EvolvedRecipe = PastaPot:10,
        FoodType    =   Pasta,
    }

    recipe Place Fresh Pasta in Cooking Pot
	{
	   PastaMachine.FreshPasta;10,
	   destroy WaterPot,

	   Result:WaterPotPasta,
	   Time:50.0,
	   OnCreate:Recipe.OnCreate.AddBaseIngredientToCookingVessel,
	   Category:Cooking,
	   OnGiveXP:Recipe.OnGiveXP.None,
	}
}
